create database if not exists clinica_vet_pp;
use clinica_vet_pp;

create table if not exists tutor (
    id bigint primary key auto_increment,
    nome_completo varchar(120) not null,
    cpf varchar(20),
    telefone varchar(20),
    email varchar(120),
    endereco varchar(150),
    cidade varchar(80),
    observacoes varchar(255)
);

create table if not exists pet (
    id bigint primary key auto_increment,
    nome varchar(80) not null,
    especie varchar(60) not null,
    raca varchar(80),
    sexo varchar(20),
    cor varchar(40),
    data_nascimento date,
    peso decimal(10,2),
    porte varchar(30),
    castrado boolean,
    numero_microchip varchar(60),
    alergias varchar(255),
    status_vacinal varchar(80),
    tutor_id bigint not null,
    constraint fk_pet_tutor foreign key (tutor_id) references tutor(id) on delete cascade
);

create table if not exists prontuario (
    id bigint primary key auto_increment,
    pet_id bigint not null unique,
    historico_clinico varchar(255),
    medicacao_continua varchar(255),
    restricoes_alimentares varchar(255),
    nivel_prioridade varchar(30),
    ultima_atualizacao datetime,
    constraint fk_prontuario_pet foreign key (pet_id) references pet(id) on delete cascade
);

create table if not exists consulta (
    id bigint primary key auto_increment,
    pet_id bigint not null,
    data_consulta datetime not null,
    tipo_consulta varchar(80) not null,
    status_consulta varchar(40),
    observacoes varchar(255),
    valor decimal(10,2),
    constraint fk_consulta_pet foreign key (pet_id) references pet(id) on delete cascade
);
